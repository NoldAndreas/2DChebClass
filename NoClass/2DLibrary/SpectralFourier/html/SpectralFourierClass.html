classdef (Abstract) SpectralFourierClass < Shape

    %**********************************************
    %************   Constructor   *****************
    %**********************************************
    methods
        function this = SpectralFourierClass(N1,N2)
             this@Shape(N1,N2);
             if(mod(N2,2) == 1)
                 exception = MException('SpectralFourierClass:Constructor','N2 has to be even');
                 throw(exception);
             end

             this.Pts.x1 = ClenCurtFlip(N1-1);
             this.Pts.x2 = (0:(N2-1))'/(N2);
        end

    end

    %**********************************************
    %************   Initializations    *************
    %**********************************************
    methods (Access = public)
        function this = Initialization(this)
            [y1,dy1]        = this.PhysSpace1(this.Pts.x1);
            [y2,dy2]        = this.PhysSpace2(this.Pts.x2);
            this.Pts.y1_kv  = kron(y1,ones(size(y2)));
            this.Pts.y2_kv  = kron(ones(size(y1)),y2);

%             this.Diff       = GetDifferentiationMatrix();
%             this.Int        = GetIntegrationVector(this);%,dy1,dy2);
%             this.Ind        = GetIndicesBox(x1,x2);
        end

        function M  = GetIntegrationVector(this)
            [~,dy1]        = this.PhysSpace1(this.Pts.x1);
            [~,dy2]        = this.PhysSpace2(this.Pts.x2);

            [~,wInt1] = ClenCurtFlip(this.N1-1);
            %1/N2 weight of integration in second direction
            M = kron(dy1'.*wInt1,(dy2').*ones(1,this.N2))/this.N2;
            this.Int = M;
        end

        function Diff = GetDifferentiationMatrix(this,x1)

            if(nargin == 1)
                x1 = this.Pts.x1;
            end

            N2 = this.N2;

            Diff1 = barychebdiff(x1,3);
            Diff2 = struct();
            %[Dx1,DDx1,DDDx1]= barychebdiff(x1);

            %Differentiation matrix in x2
            h         = 2*pi/N2;
            column    = [0 .5*(-1).^(1:N2-1).*cot((1:N2-1)*h/2)];
            Diff2.Dx  = toeplitz(column,column([1 N2:-1:2]))*2*pi;% 2pi is needed to go back to the -1;1 - domain


            h         = 2*pi/N2;
            column2   = [(-(pi^2)/(3*h^2) - 1/6)  -.5*(-1).^(1:N2-1)./((sin((1:N2-1)*h/2)).^2)];
            Diff2.DDx = toeplitz(column2,column2([1 N2:-1:2]))*(2*pi)^2;


            Sel = {'Dy1' ;'DDy1' ; 'Dy2'; 'DDy2' ;...
                  'Dy1Dy2'; 'Lap' ;'grad' ;'div';...
                  'gradDiv'; 'LapVec'};

            Maps = struct('PhysSpace1',@PhysSpace1,...
                          'PhysSpace2',@PhysSpace2,...
                          'CompSpace1',@CompSpace1,...
                          'CompSpace2',@CompSpace2);

            Diff = PhysicalDerivatives(Maps,Pts,Sel,Diff1,Diff2,2);
            %Dx1,Dx2,DDx1,DDx2);

        end

    end




end
